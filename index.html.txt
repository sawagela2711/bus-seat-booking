<h2>🚌 Выберите место</h2>

<div class="bus" id="bus"></div>

<button onclick="confirmSeat()">Подтвердить</button>

<script>
  const bus = document.getElementById("bus");

  let selectedSeat = null;

  // список занятых мест (пример)
  const bookedSeats = ["A2", "C3", "F1"];

  const rows = 12;

  for (let i = 0; i < rows; i++) {
    const row = document.createElement("div");
    row.className = "row";

    const rowLetter = String.fromCharCode(65 + i); // A, B, C...

    for (let j = 1; j <= 4; j++) {

      if (j === 3) {
        const aisle = document.createElement("div");
        aisle.className = "aisle";
        row.appendChild(aisle);
      }

      const seatId = rowLetter + j;

      const seat = document.createElement("div");
      seat.className = "seat";
      seat.innerText = seatId;

      if (bookedSeats.includes(seatId)) {
        seat.classList.add("booked");
      }

      seat.onclick = () => {
        if (seat.classList.contains("booked")) return;

        document.querySelectorAll(".seat").forEach(s => s.classList.remove("selected"));

        seat.classList.add("selected");
        selectedSeat = seatId;
      };

      row.appendChild(seat);
    }

    bus.appendChild(row);
  }

  // добавим 49-е место
  const extra = document.createElement("div");
  extra.className = "seat";
  extra.innerText = "Z1";

  extra.onclick = () => {
    document.querySelectorAll(".seat").forEach(s => s.classList.remove("selected"));
    extra.classList.add("selected");
    selectedSeat = "Z1";
  };

  bus.appendChild(extra);

  function confirmSeat() {
    if (!selectedSeat) {
      alert("Выберите место");
      return;
    }

    if (window.ManyChat) {
      ManyChat.setCustomField("seat", selectedSeat);
      ManyChat.close();
    } else {
      alert("Вы выбрали: " + selectedSeat);
    }
  }
</script>
